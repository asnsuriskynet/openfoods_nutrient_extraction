def extract_valid_images(entry: Union[List[Any], np.ndarray, None]) -> List[dict]:
    """
    Extract valid image dictionaries from a given list or array of image data.
    Valid image is a dictionary where imgid key is non-Nan.

    Args:
        entry (Union[List[Any], np.ndarray, None]): List or array containing image info dictionaries.

    Returns:
        List[dict]: A list of valid image dictionaries.
    """
    # Return empty list if None or not iterable
    if entry is None or not isinstance(entry, Iterable):
        return []

    # Convert NumPy array to list
    if isinstance(entry, np.ndarray):
        entry = entry.tolist()

    # Filter valid dictionaries with a non-None 'imgid' + Return
    return [img for img in entry if isinstance(img, dict) and img.get('imgid') is not None]



def build_off_image_url(code: str, img: Dict[str, any], size: str = "400") -> str:
    """
    Construct image URL based on the product code and image metadata.

    Args:
        code (str): The product barcode as a string.
        img (Dict[str, Any]): A dictionary containing image metadata, expected to include 'key', 
                              'rev'(revision number), 'sizes' (optional)
        size (str, optional): Desired image size (e.g., '100', '200', '400', 'full'). Defaults == '400'.

    Returns:
        str: A URL string pointing to the image file on OpenFoodFacts.
    """
    # Pad code to 13 digits (Standard)
    code = code.zfill(13)

    # Create nested folder structure (3 + 3 + 3 + 4 digits) 
    # 1234567890123 ---> ["123", "456", "789", "0123"]
    folders = [code[i:i+3] for i in range(0, 9, 3)] + [code[9:]]

    # "123/456/789/0123"
    folder_path = "/".join(folders)

    key = img.get("key")
    rev = int(img.get("rev")) if img.get("rev") is not None else 1

    # Ger 400 image size or fallback to available size
    available_sizes = img.get("sizes", {}).keys()
    if size not in available_sizes:
        if "400" in available_sizes:
            size = "400"
        else:
            size = next(iter(available_sizes), "400")

    url = f"https://static.openfoodfacts.org/images/products/{folder_path}/{key}.{rev}.{size}.jpg"
    return url


def extract_image_urls(code: Union[str, int], images: Union[List[dict], np.ndarray, None], size: str = "400") -> Dict[str, str]:
    """
    Extract URLs of product images from OpenFoodFacts metadata.

    Args:
        code (Union[str, int]): The product's barcode (numeric string or integer).
        images (Union[List[dict], np.ndarray, None]): A list or array of image metadata dictionaries.
        size (str, optional): Desired image size (e.g., '100', '400', 'full'). Defaults to '400'.

    Returns:
        Dict[str, str]: A dictionary mapping image 'key' (e.g., 'front', 'ingredients') to image URLs.
    """

    # if not iterable, return empty dict
    if not hasattr(images, '__iter__'):
        return {}

    # Convert NumPy array to list 
    if isinstance(images, np.ndarray):
        images = images.tolist()

    # iterate through images, build URLs, return URL dictionary
    url_dict = {}
    for img in images:
        if isinstance(img, dict) and img.get("imgid") is not None:
            key = img.get("key")
            url = build_off_image_url(str(code), img, size=size)
            if key:
                url_dict[key] = url

    return url_dict


def extract_image_counts(row: Any) -> Dict[str, int]:
    """
    Count number of front, nutrition, and ingredient images in a given product image metadata dictionary.

    Args:
        row (Any): A dictionary containing image keys (e.g., 'front_en', 'nutrition_fr'), 
                   or any other types (Wont contribute to count).

    Returns:
        Dict[str, int]: A dictionary with counts of front, nutrition, and ingredient-related images.
                        Keys: 'front_images', 'nutrition_images', 'ingredient_images'.
    """
    
    if isinstance(row, dict):
        keys = row.keys()
        return {
            'front_images': sum('front' in k for k in keys),
            'nutrition_images': sum('nutrition' in k for k in keys),
            'ingredient_images': sum('ingredient' in k for k in keys)
        }

    return {'front_images': 0, 'nutrition_images': 0, 'ingredient_images': 0}


def extract_nutrient(nutrient_array: Union[np.ndarray, List[Dict[str, Any]], None], name: str) -> float:
    """
    Extract the nutrient value per 100g from nutrient array for nutrient name.

    Args:
        nutrient_array (Union[np.ndarray, List[Dict[str, Any]], None]): Array or list of nutrient dictionaries.
            Each dictionary should have at least 'name' and '100g' keys.
        name (str): The name of the nutrient to extract (e.g., 'salt', 'sugars', 'proteins').

    Returns:
        float: The nutrient value per 100g if found; otherwise, np.nan.
    """
    if isinstance(nutrient_array, (np.ndarray, list)):
        for item in nutrient_array:
            if item.get('name') == name:
                return item.get('100g', np.nan)
    return np.nan


def print_availability_stats(df: pd.DataFrame, attributes: List[str], countries: List[str]) -> None:
    """
    Print the percentage of products available with specified attributes per country.

    For each attribute and country, function calculates and prints the percentage of products
    that have the attribute available (attr == 1) among the products for that country
    (country_product == 1).

    Args:
        df (pd.DataFrame): DataFrame containing product data including attribute columns and country product flags.
        attributes (List[str]): List of attribute column names to check availability (e.g. 'has_ingredient_text').
        countries (List[str]): List of country identifiers matching suffixes in country product columns (e.g. ['france', 'us']).

    Returns:
        None: This function only prints results and does not return a value.
    """
    for attr in attributes:
        print(f"--- {attr} ---")
        for country in countries:
            country_col = f"{country}_product" 
            mask = (df[attr] == 1) & (df[country_col] == 1)
            country_total = len(df[df[country_col] == 1])
            pct = len(df[mask]) / country_total if country_total > 0 else 0
            print(f"{country.capitalize()}: {pct:.2%} of products have {attr} available")
        print()


def compute_attribute_availability_by_country(
    df: pd.DataFrame,
    attributes: List[str],
    countries: List[str]
) -> pd.DataFrame:
    """
    Compute the percentage of products that have specific attributes available for each country.

    Args:
        df (pd.DataFrame): The input DataFrame containing attribute and country columns.
        attributes (List[str]): List of column names representing the attributes to check (e.g., 'has_ingredient_text').
        countries (List[str]): List of country identifiers that match the '<country>_product' column names.

    Returns:
        pd.DataFrame: A pivot table with attributes as rows, countries as columns, and values as percentage availability.
    """
    records: List[Dict[str, object]] = []

    for attr in attributes:
        for country in countries:
            country_col = f"{country}_product"  # e.g., 'france_product'
            country_mask = df[country_col] == 1
            attr_mask = df[attr] == 1
            combined_mask = country_mask & attr_mask

            country_total = country_mask.sum()
            pct = combined_mask.sum() / country_total if country_total > 0 else 0.0

            records.append({
                'attribute': attr,
                'country': country.capitalize(),
                'percent': pct
            })

    summary_df = pd.DataFrame(records)
    return summary_df.pivot(index='attribute', columns='country', values='percent')


def us_france_box_plot_comparison(df: pd.DataFrame, column_to_compare: str) -> None:
    """
    Plot side-by-side boxplots comparing numeric column between 
    French and US products, and print summary statistics.

    Args:
        df (pd.DataFrame): Input DataFrame containing at least 'france_product', 'us_product', and the column to compare.
        column_to_compare (str): Name of the column to visualize and compare (must be numeric).

    Returns:
        None
    """
    france_data = df[df['france_product'] == 1][column_to_compare].dropna()
    us_data = df[df['us_product'] == 1][column_to_compare].dropna()

    # Optional Outlier removal
    france_data = france_data[france_data < 200]
    us_data = us_data[us_data < 200]

    # Plot
    plt.figure(figsize=(12, 5))

    plt.subplot(1, 2, 1)
    sns.boxplot(y=france_data)
    plt.title(f'{column_to_compare} Distribution - France')
    plt.ylabel(column_to_compare)

    plt.subplot(1, 2, 2)
    sns.boxplot(y=us_data)
    plt.title(f'{column_to_compare} Distribution - US')
    plt.ylabel(column_to_compare)

    plt.tight_layout()
    plt.show()

    # Summary stats
    print("\nSummary Statistics:")
    print("\nFrance:")
    print(france_data.describe(percentiles=[.25, .5, .75]).round(2))

    print("\nUnited States:")
    print(us_data.describe(percentiles=[.25, .5, .75]).round(2))


def plot_text_vs_image_availability(
    df: pd.DataFrame,
    text_col: str,
    image_col: str,
    title_prefix: str = '',
    colors: Union[list, tuple] = ('skyblue', 'orange')) -> None:
    """
    Plots a grouped bar chart comparing text and image availability.

    Args:
        df (pd.DataFrame): The input DataFrame.
        text_col (str): Column name indicating whether text data is available.
        image_col (str): Column name indicating whether image data is available
        title_prefix (str, optional): Title prefix for the plot. Example: 'Ingredient' or 'Nutrition'.
        colors (list or tuple, optional): Colors for the bars in the plot. Default is ('skyblue', 'orange').

    Returns:
        None: Displays a matplotlib plot.
    """

    #Groupby, Map Values
    comparison = df.groupby([text_col, image_col]).size().reset_index(name='count')
    comparison['Text'] = comparison[text_col].map({True: 'Text Available', False: 'Text Missing', 1: 'Text Available', 0: 'Text Missing'})
    comparison['Image'] = comparison[image_col].map({True: 'Image Available', False: 'Image Missing', 1: 'Image Available', 0: 'Image Missing'})

    pivot = comparison.pivot(index='Text', columns='Image', values='count').fillna(0)

    #Plot
    pivot.plot(kind='bar', figsize=(8, 5), color=colors)
    plt.title(f'{title_prefix} Text vs. {title_prefix} Image Availability', pad=15)
    plt.ylabel('Number of Products')
    plt.xlabel(f'{title_prefix} Text Availability')
    plt.xticks(rotation=0)
    plt.legend(title=f'{title_prefix} Image')
    plt.tight_layout()
    plt.show()


def plot_binary_availability_comparison(
    df: pd.DataFrame,
    col_a: str,
    col_b: str,
    label_a: str = 'Column A',
    label_b: str = 'Column B',
    colors: Union[list, tuple] = ('skyblue', 'orange')
) -> None:
    """
    Plots a grouped bar chart comparing two binary availability columns.

    Args:
        df (pd.DataFrame): The input DataFrame.
        col_a (str): First binary column (used as index in plot).
        col_b (str): Second binary column (used as column/legend in plot).
        label_a (str): Label for the first column (e.g., 'Text', 'Nova Group').
        label_b (str): Label for the second column (e.g., 'Image', 'Ingredient Text').
        colors (list or tuple): Bar colors.

    Returns:
        None. Displays the matplotlib plot.
    """

    # Group and count, map
    comparison = df.groupby([col_a, col_b]).size().reset_index(name='count')
    comparison[label_a] = comparison[col_a].map({True: f'{label_a} Available', False: f'{label_a} Missing', 1: f'{label_a} Available', 0: f'{label_a} Missing'})
    comparison[label_b] = comparison[col_b].map({True: f'{label_b} Available', False: f'{label_b} Missing', 1: f'{label_b} Available', 0: f'{label_b} Missing'})

    pivot = comparison.pivot(index=label_a, columns=label_b, values='count').fillna(0)

    # Plot
    pivot.plot(kind='bar', figsize=(8, 5), color=colors)
    plt.title(f'{label_a} vs {label_b} Availability', pad=15)
    plt.ylabel('Number of Products')
    plt.xlabel(f'{label_a} Availability')
    plt.xticks(rotation=0)
    plt.legend(title=f'{label_b} Availability')
    plt.tight_layout()
    plt.show()


def plot_binary_column_pie(df: pd.DataFrame, column: str) -> None:
    """
    Plots a pie chart showing the distribution of rows based on binary column.

    Parameters:
    - df: pd.DataFrame - DataFrame containing the data.
    - column: str - Name of the binary column to analyze.

    Returns:
    - None
    """
    counts = df[column].value_counts().sort_index()

    base_label = column.replace('has_', '').replace('_', ' ').title()
    labels = [f'Products without {base_label}', f'Products with {base_label}']

    data = [counts.get(0, 0), counts.get(1, 0)]

    plt.figure(figsize=(5, 5))
    plt.pie(data, labels=labels, autopct='%1.1f%%', colors=['#ff9999','#66b3ff'], startangle=140)
    plt.title(f'Distribution of Products by {column}', pad=20)
    plt.axis('equal')
    plt.show()

def extract_nutrition_url(image_dict: dict) -> str:
    """
    Extracts the first URL from a dictionary where the key contains 'nutrition'.
    
    Parameters:
    - image_dict: dict or NaN
    
    Returns:
    - str or None
    """
    if isinstance(image_dict, dict):
        for key, value in image_dict.items():
            if 'nutrition' in key.lower():
                return value
    return None
